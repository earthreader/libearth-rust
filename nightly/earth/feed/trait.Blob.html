fn sanitized_html<'a>(&'a self, base_uri: Option<&'a str>) -> Box<Display + 'a>

Get the secure HTML string of the text. If it's a plain text, this returns entity-escaped HTML string, if it's a HTML text, value is sanitized, and if it's a binary data, this returns base64-encoded string.

let text = Text::text("<Hello>");
let html = Text::html("<script>alert(1);</script><p>Hello</p>");
assert_eq!(format!("{}", text.sanitized_html(None)), "&lt;Hello&gt;");
assert_eq!(format!("{}", html.sanitized_html(None)), "<p>Hello</p>");
